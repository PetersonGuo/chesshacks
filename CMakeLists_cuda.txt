cmake_minimum_required(VERSION 3.18...3.27)
project(chesshacks LANGUAGES CXX)

# Check if CUDA is available
include(CheckLanguage)
check_language(CUDA)

# Enable CUDA if available
if(CMAKE_CUDA_COMPILER)
    enable_language(CUDA)
    set(CUDA_ENABLED ON)
    message(STATUS "CUDA found - enabling GPU acceleration")
    
    # Set CUDA architecture (adjust based on your GPU)
    # Common values: 50 (Maxwell), 60 (Pascal), 70 (Volta), 75 (Turing), 80 (Ampere), 89 (Ada)
    if(NOT DEFINED CMAKE_CUDA_ARCHITECTURES)
        set(CMAKE_CUDA_ARCHITECTURES 75 80 86 89)  # Support modern GPUs
    endif()
    
    # Set CUDA standard
    set(CMAKE_CUDA_STANDARD 14)
    set(CMAKE_CUDA_STANDARD_REQUIRED ON)
else()
    set(CUDA_ENABLED OFF)
    message(STATUS "CUDA not found - GPU acceleration disabled")
endif()

if (CMAKE_VERSION VERSION_LESS 3.18)
    set(DEV_MODULE Development)
else()
    set(DEV_MODULE Development.Module)
endif()
find_package(Python 3.8 COMPONENTS Interpreter ${DEV_MODULE} REQUIRED)

execute_process(
    COMMAND "${Python_EXECUTABLE}" -m nanobind --cmake_dir
    OUTPUT_STRIP_TRAILING_WHITESPACE
    OUTPUT_VARIABLE nanobind_ROOT
)
find_package(nanobind CONFIG REQUIRED)

# Add the source directory to include paths
include_directories(${CMAKE_SOURCE_DIR}/src)

# Source files
set(SOURCE_FILES
    src/bindings.cpp
    src/chess_board.cpp
    src/cuda_utils.cpp
    src/move_ordering.cpp
    src/transposition_table.cpp
    src/evaluation.cpp
    src/opening_book.cpp
    src/search_and_utils.cpp
)

# Add CUDA source if available
if(CUDA_ENABLED)
    list(APPEND SOURCE_FILES src/cuda_eval.cu)
    
    # Create the extension module with CUDA support
    nanobind_add_module(c_helpers ${SOURCE_FILES})
    
    # Set CUDA compilation flags
    target_compile_definitions(c_helpers PRIVATE CUDA_ENABLED)
    set_target_properties(c_helpers PROPERTIES
        CUDA_SEPARABLE_COMPILATION ON
        CUDA_RESOLVE_DEVICE_SYMBOLS ON
    )
else()
    # Create the extension module without CUDA
    nanobind_add_module(c_helpers ${SOURCE_FILES})
endif()

# Create a symlink from build/tests to ../tests for easier test execution
add_custom_command(TARGET c_helpers POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E create_symlink
        ${CMAKE_SOURCE_DIR}/tests
        ${CMAKE_BINARY_DIR}/tests
    COMMENT "Creating symlink to tests directory"
)
